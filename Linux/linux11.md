# Linux常用命令

### 一、入门

```shell
#历史命令
history
#历史命令向上切换
ctrl + p
#历史命令向下切换
ctrl + n

#光标切换
#直接到行首
ctrl + a
#直接到行尾
ctrl + e
#光标向左
ctrl + b
#光标向右
ctrl + f

#字符删除
#删除光标前面的第一个字符
ctrl + h
#删除光标后面的第一个字符(删除光标所覆盖的字符)
ctrl + d
#删除光标前边所有的字符
ctrl + u

#自动补全文件名和路径
tab
#查看日期+当前时间
date
#清屏
ctrl + L
#切换目录
cd
#停止
ctrl + c
```

## 二、小技巧

```shell
#用户目录
#绝对路径
cd /home/.../...
#相对路径
cd ./  						 #当前目录
cd ../ 						 #上一级目录
cd -						 # 两个临近的目录之间的切换
cd ~						 #切换到家目录
[root@test1 bin]#
root			#登录当前用户名
test1			#主机名
bin				#当前所在目录
#				#管理员
$				#普通用户
#查看当前目录
pwd
```

## 三、文件颜色<------>文件属性

```shell
tree /home    #查看/home下面的树状结构
#颜色					文件类型
#白色	<---------------> 普通文件	 【-】
#蓝色	<---------------> 目录	   【d】
#绿色	<---------------> 可执行文件	
#红色	<---------------> 压缩文件	
#青色	<---------------> 链接文件	【l】(window下面的快捷方式)
#黄色	<---------------> 设备文件(块设备【b】、字符设备【c】、socket文件【s】、管道文件【p】)
#灰色	<---------------> 其他文件
```

## 四、ls

```shell
ls	/home		#列出当前目录
ls -a /home		#列出包括隐藏的文件
ls -l /home		#列出文件的详细信息
ls -la /home		#列出包括隐藏文件的详细信息
```

## 五、文件信息

```shell
drwx------.  2 hxlShare hxlShare 4096 1月  11 19:30 hxlShare
d			#文件类型【-】【d】【l】【p】【s】【b】【c】
r			#可读
w			#可写
x			#可执行
2			#文件的硬连接数
hxlShare	#该文件或者目录的所有者
hxlShare	#该文件和目录的所有组
4096		#占用的存储空间
1月		   #文件最后创建时间或修改时间
hxlShare	#文件名
#d之后的1-3位字符			所有者
#d之后的4-6位字符		    同组用户
#d之后的7-9位字符		    其他人
```

## 六、文件目录操作

```shell
#############################创建目录##############################
mkdir newshelen		#创建目录
mkdir -p /aa/bb/cc	或者 mkdir /aa/bb/cc -p	#创建符合目录
mkdir -v /aa	#显示创建信息
mkdir -pv /aa/xx/aa 
#############################删除##################################
rmdir /aa		#删除空目录
rm -r /aa		#递归删除目录及其内容
rm -i /aa		#删除前确认
rm -I /aa		#超过三个文件删除确认
rm -f /aa		#强制删除
#############################创建文件##############################
touch hanxinlong.php
如果创建的文件不存在，创建文件，如果文件存在修改文件的创建时间
#############################拷贝文件##############################
cp -R /a/b/c /d/		#递归循环复制
#############################查看文件内容##########################
cat 1.php			#查看文件
#分屏显示
more 1.php
#按回车键一行一行显示
#按空格键一页一页显示
#退出	q键
less 1.php
#按回车键一行一行显示
ctrl + p #向上显示(一行)
ctrl + b #向后显示(一行)
head 1.php #显示该文件默认前10行的内容
head -5 1.php #显示该文件默认前5行的内容
tail 1.php #显示该文件默认后10行的内容
tail -5 1.php #显示该文件默认后5行的内容
#############################移动文件##############################
mv 1.php 2.php		#修改文件名
mv 1.php /hanxinlong #移动文件
#############################创建软链接############################
ln -s /root/hello(必须为绝对路径) hellowrod
#############################创建硬链接############################
ln  hello hellowrod
#############################查看文件行数##########################
wc -l 1.log		#统计输出行数
wc -m 1.log		#统计输出字符数
wc -c 1.log		#统计输出自己数
du -h			#查看当前目录的目录大小
df -h			#查看磁盘使用情况
#############################which############################
which #查看当前命令的路径
whomai #查看当前用户
#############################文件权限############################
#修改文件权限
#  1> 文件设定法 chmod [who] [+|-|=] [model]
[who] 			#文件所有者 u
				#文件所属组 g
				#其他人	 o
				#所有人	 a  一般默认为a
[+] 			#添加权限
[-] 			#减少权限
[=] 			#覆盖原来的权限
[model]: 		# r 可读
				# w 可写
				# x 可执行
#  2> 数字设定法 
	r 4， w 2, x 1.
#  3>改变文件或目录的所有者或所有组
chown www:root 1.php
chgrp www 1.php		#修改用户组
#############################文件查找############################
#按照文件名查找
find + [查找的目录] + [-name] + [文件的名字(*|？)]
eg: find /home/www -name hanxinlong
#文件大小查找
find + [查找的目录] + [-size] + [数字]
eg: find + /etc/ -size -10k  #查找小于10k的文件
eg: find + /etc/ -size +10m  #查找大于10m的文件
eg：find + /etc/ -size +10m -size -100m
#文件类型查找
find + [查找的目录] + [-type] [类型(d/f/b/c/s/p/l)]
eg: find /home/ -type d
#############################文件检索############################
grep -r [查找的内容] + [查找的目录]
eg: grep -r h /etc/
```
